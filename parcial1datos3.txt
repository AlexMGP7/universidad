Ack(m,n):
    init-stack(s)
    push(m,s)
    while not empty?(s) do
        m := top(s)
        pop(s)
        if m=0 then
                n := n+1
        else
        if  n=0 then
                n := 1
                push(m-1,s)
        else
                n := n-1
                push(m-1,s)
                push(m,s)
    return(n)
